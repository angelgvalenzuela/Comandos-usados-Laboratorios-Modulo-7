    read only = no

COMANDOS USADOS PRÁCTICA 3: CREACIÓN DE CONTROLADOR DE DOMINIO CON CLIENTE WINDOWS 
# 1. Configurar Hostname
sudo hostnamectl set-hostname dc1.so3.inet
sudo nano /etc/hosts
# Agregar esta línea (Cambia la IP por la de tu servidor Linux actual):
192.168.1.109   dc1.so3.inet   dc1

# 2. Actualizar sistema
sudo dnf -y update

# 3. Habilitar Repositorio EPEL 10 y CRB
sudo dnf install https://dl.fedoraproject.org/pub/epel/epel-release-latest-10.noarch.rpm -y
sudo crb enable
sudo dnf makecache

# 4. Instalación de Dependencias de compilación
sudo dnf install -y wget tar gcc make python3-devel \
flex bison gdb git keyutils-libs-devel \
libacl-devel libattr-devel libblkid-devel \
libxml2-devel libxslt-devel \
openldap-devel pam-devel popt-devel \
readline-devel zlib-devel systemd-devel \
lmdb-devel jansson-devel gpgme-devel \
libarchive-devel libicu-devel \
perl-ExtUtils-MakeMaker perl-Parse-Yapp \
rpcgen libtirpc-devel

# 5. Instalar librerías de criptografía y seguridad
sudo dnf install -y gnutls-devel python3-cryptography python3-dns


## Descarga y Compilación de Samba ##

# 6. Crear directorio, descargar y descomprimir
sudo mkdir -p /samba
cd /samba
sudo wget https://download.samba.org/pub/samba/stable/samba-4.21.1.tar.gz
sudo tar -zxvf samba-4.21.1.tar.gz
cd samba-4.21.1/

# 7. Configurar dependencias Perl extra
sudo dnf install perl-Parse-Yapp perl-Test-Simple perl-ExtUtils-MakeMaker -y
sudo dnf install perl-FindBin perl-lib perl-English -y

# 8. Configurar compilación, Compilar e Instalar
sudo dnf install -y dbus-devel
sudo dnf install -y python3-markdown
sudo ./configure --prefix=/usr/local/samba --enable-fhs --with-systemd --with-ads --with-ldap
sudo make
sudo make install

# 9. Configuración del Entorno (Variables PATH)
export PATH=/usr/local/samba/bin:/usr/local/samba/sbin:$PATH
echo 'export PATH=$PATH:/usr/local/samba/bin:/usr/local/samba/sbin' >> ~/.bashrc
source ~/.bashrc


## Aprovisionamiento del Dominio ##

# 10. Crear el dominio 
sudo /usr/local/samba/bin/samba-tool domain provision \
--use-rfc2307 \
--interactive \
--option="interfaces= lo ens33" \
--option="bind interfaces only=yes"

# Realm: SO3.inet
# Domain: SO3-DOM
# Server Role: dc
# DNS backend: SAMBA_INTERNAL
# DNS Forwarder IP address: 192.168.1.109
# Administrator password: 20251447Av

## 10.1. Ajustes de Seguridad y DNS (Post-Aprovisionamiento) ##

# 1. Desactivar SELinux para permitir que Samba responda consultas DNS
sudo setenforce 0

# 2. Crear registros DNS A manualmente (Soluciona el error de "Non-existent domain")
# Registrar el dominio raíz:
sudo /usr/local/samba/bin/samba-tool dns add 127.0.0.1 so3.inet @ A 192.168.1.109

# Registrar el nombre específico del servidor:
sudo /usr/local/samba/bin/samba-tool dns add 127.0.0.1 so3.inet angelserver1 A 192.168.1.109

# 3. Reiniciar el servicio para aplicar todos los cambios
sudo systemctl restart samba-ad-dc

# 4. Verificación final (Desde Windows o Linux):
# nslookup so3.inet

# 11. Editar el archivo de resolución DNS
sudo nano /etc/resolv.conf
# Borrar lo que tenga y poner solo esto:
nameserver 127.0.0.1
nameserver 8.8.8.8


## Configuración de Kerberos y Systemd ##

# 12. Copiar configuración de Kerberos
sudo cp /usr/local/samba/var/lib/samba/private/krb5.conf /etc/krb5.conf

# 13. Crear el servicio para que inicie automático (Ya con las correcciones hechas)
sudo nano /etc/systemd/system/samba-ad-dc.service

# Pegar esto dentro del archivo:
[Unit]
Description=Samba Active Directory Domain Controller
After=network.target remote-fs.target nss-lookup.target

[Service]
Type=simple
ExecStart=/usr/local/samba/sbin/samba -i
ExecReload=/bin/kill -HUP $MAINPID

[Install]
WantedBy=multi-user.target


# 14. Iniciar el servicio y apagar SELinux
sudo setenforce 0
sudo systemctl daemon-reload
sudo systemctl enable --now samba-ad-dc

# 15. Configuración del Firewall (Opcional)
sudo firewall-cmd --permanent --add-service={dns,kerberos,ldap,ldaps}
sudo firewall-cmd --permanent --add-port={53,88,135,139,389,445,464,636,3268,3269}/tcp
sudo firewall-cmd --permanent --add-port={53,88,137,138,389,464}/udp
sudo firewall-cmd --reload
